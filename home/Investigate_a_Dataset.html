def slope(x_val,y_val):
    x = np.array(x_val)
    y = np.array(y_val)
    m=((np.mean(x)*np.mean(y))-np.mean(x*y)/((np.mean(x)*np.mean(x))-np.mean(x*x)))
    m = round(m,2)
    b = (np.mean(y)-np.mean(x)*m)
    b = round(b,2)
    return m,b
